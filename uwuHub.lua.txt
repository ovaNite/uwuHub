-- Slider Logic
local Dragging = false
local function UpdateSlider()
    local Percent = math.clamp((UserInputService:GetMouseLocation().X - SliderBg.AbsolutePosition.X) / SliderBg.AbsoluteSize.X, 0.01, 1)
    Settings.Smoothness = Percent SliderFill.Size = UDim2.new(Percent, 0, 1, 0) SliderLabel.Text = "Smoothness: " .. string.format("%.2f", Percent)
end
SliderBg.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then Dragging = true UpdateSlider() end end)
UserInputService.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then Dragging = false end end)
UserInputService.InputChanged:Connect(function(i) if Dragging and i.UserInputType == Enum.UserInputType.MouseMovement then UpdateSlider() end end)

--- 👁️ ESP SYSTEM ---
local function CreateVisuals(Player)
    local Line = Drawing.new("Line")
    Line.Color = Color3.fromRGB(255, 0, 0) -- Rote Tracers
    Line.Thickness = 1
    local Box = Drawing.new("Square")
    RunService.RenderStepped:Connect(function()
        if Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") and Player ~= LocalPlayer then
            local highlight = Player.Character:FindFirstChild("uwuCharms")
            local isEnemy = (not Settings.TeamCheck or Player.Team ~= LocalPlayer.Team)
            local Pos, OnScreen = Camera:WorldToViewportPoint(Player.Character.HumanoidRootPart.Position)

            -- Tracers vom Crosshair (Bildschirmmitte)
            if Settings.TracersEnabled and OnScreen and isEnemy then
                Line.Visible = true 
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2) 
                Line.To = Vector2.new(Pos.X, Pos.Y)
            else Line.Visible = false end

            -- Charms & Box (wie bisher)
            if Settings.CharmsEnabled and isEnemy then
                if not highlight then highlight = Instance.new("Highlight", Player.Character) end
                highlight.Enabled = true
            elseif highlight then highlight.Enabled = false end

            if Settings.ESPEnabled and OnScreen and isEnemy then
                Box.Visible = true Box.Size = Vector2.new(2000/Pos.Z, 2500/Pos.Z) Box.Position = Vector2.new(Pos.X - Box.Size.X/2, Pos.Y - Box.Size.Y/2)
            else Box.Visible = false end
        else Line.Visible = false Box.Visible = false end
    end)
end
for _, p in pairs(Players:GetPlayers()) do CreateVisuals(p) end
Players.PlayerAdded:Connect(CreateVisuals)

--- 🎯 AIMBOT ENGINE ---
RunService.RenderStepped:Connect(function()
    local IsPressed = (Settings.AimKey.EnumType == Enum.KeyCode and UserInputService:IsKeyDown(Settings.AimKey)) or (Settings.AimKey.EnumType == Enum.UserInputType and UserInputService:IsMouseButtonPressed(Settings.AimKey))
    if Settings.AimbotEnabled and IsPressed then
        local Target, MaxDist = nil, Settings.Range
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild(Settings.TargetPart) then
                if not Settings.TeamCheck or p.Team ~= LocalPlayer.Team then
                    local Pos, OnScreen = Camera:WorldToViewportPoint(p.Character[Settings.TargetPart].Position)
                    local Dist = (Vector2.new(Pos.X, Pos.Y) - UserInputService:GetMouseLocation()).Magnitude
                    if OnScreen and Dist < MaxDist then Target = p.Character[Settings.TargetPart] MaxDist = Dist end
                end
            end
        end
        if Target then Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, Target.Position), Settings.Smoothness) end
    end
end)
