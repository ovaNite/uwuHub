FOVCircle.Thickness = Settings.FOVThickness

local function GetClosestPlayer()
    local Target, MouseDist = nil, Settings.Range
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild(Settings.TargetPart) then
            if not Settings.TeamCheck or p.Team ~= LocalPlayer.Team then
                local Pos, OnScreen = Camera:WorldToViewportPoint(p.Character[Settings.TargetPart].Position)
                local D = (Vector2.new(Pos.X, Pos.Y) - UserInputService:GetMouseLocation()).Magnitude
                if OnScreen and D < MouseDist then Target = p.Character[Settings.TargetPart] MouseDist = D end
            end
        end
    end
    return Target
end

local DraggingSmooth, DraggingFOV, DraggingVis = false, false, false
SmoothBg.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then DraggingSmooth = true end end)
FOVSliderBg.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then DraggingFOV = true end end)
VisRangeBg.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then DraggingVis = true end end)
UserInputService.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then DraggingSmooth = false DraggingFOV = false DraggingVis = false end end)

RunService.RenderStepped:Connect(function()
    FOVCircle.Position = UserInputService:GetMouseLocation()
    FOVCircle.Radius = Settings.Range
    local MX = UserInputService:GetMouseLocation().X
    
    if DraggingSmooth then
        local P = math.clamp((MX - SmoothBg.AbsolutePosition.X) / SmoothBg.AbsoluteSize.X, 0.01, 1)
        Settings.Smoothness = P SmoothFill.Size = UDim2.new(P,0,1,0) SmoothLabel.Text = "Smoothness: "..string.format("%.2f", P)
    end
    if DraggingFOV then
        local P = math.clamp((MX - FOVSliderBg.AbsolutePosition.X) / FOVSliderBg.AbsoluteSize.X, 0.05, 1)
        Settings.Range = math.floor(P * 500) FOVSliderFill.Size = UDim2.new(P,0,1,0)
    end
    if DraggingVis then
        local P = math.clamp((MX - VisRangeBg.AbsolutePosition.X) / VisRangeBg.AbsoluteSize.X, 0.05, 1)
        Settings.VisualRange = math.floor(P * 2000) VisRangeFill.Size = UDim2.new(P,0,1,0) VisRangeLabel.Text = "Visuals Range: "..Settings.VisualRange
    end

    local IsPressed = (Settings.AimKey.EnumType == Enum.KeyCode and UserInputService:IsKeyDown(Settings.AimKey)) or (Settings.AimKey.EnumType == Enum.UserInputType and UserInputService:IsMouseButtonPressed(Settings.AimKey))
    if Settings.AimbotEnabled and IsPressed then
        local T = GetClosestPlayer()
        if T then
            if Settings.AimbotType == "Camera" then
                Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, T.Position), Settings.Smoothness)
            else
                local Pos, _ = Camera:WorldToViewportPoint(T.Position)
                local MousePos = UserInputService:GetMouseLocation()
                mousemoverel((Pos.X - MousePos.X) * Settings.Smoothness, (Pos.Y - MousePos.Y) * Settings.Smoothness)
            end
        end
    end
end)

local function CreateESP(Player)
    local Line = Drawing.new("Line") Line.Color = Color3.fromRGB(255, 0, 0)
    local Box = Drawing.new("Square")
    RunService.RenderStepped:Connect(function()
        if Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") and Player ~= LocalPlayer then
            local isEnemy = (not Settings.TeamCheck or Player.Team ~= LocalPlayer.Team)
            local root = Player.Character.HumanoidRootPart
            local dist = (root.Position - Camera.CFrame.Position).Magnitude
            local Pos, OnScreen = Camera:WorldToViewportPoint(root.Position)
            local inRange = (dist <= Settings.VisualRange)
            local highlight = Player.Character:FindFirstChild("uwuCharms")

            if Settings.CharmsEnabled and isEnemy and inRange then
                if not highlight then highlight = Instance.new("Highlight", Player.Character) highlight.Name = "uwuCharms" end
                highlight.Enabled = true
            elseif highlight then highlight.Enabled = false end

            if Settings.TracersEnabled and OnScreen and isEnemy and inRange then
                Line.Visible = true Line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2) Line.To = Vector2.new(Pos.X, Pos.Y)
            else Line.Visible = false end

            if Settings.ESPEnabled and OnScreen and isEnemy and inRange then
                Box.Visible = true Box.Size = Vector2.new(2000/Pos.Z, 2500/Pos.Z) Box.Position = Vector2.new(Pos.X - Box.Size.X/2, Pos.Y - Box.Size.Y/2)
            else Box.Visible = false end
        else Line.Visible = false Box.Visible = false end
    end)
end
